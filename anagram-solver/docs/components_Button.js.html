import { el } from '../utils/dom.js';

/**
 * Creates a button UI element.
 *
 * @param {string} text - Button label.
 * @param {Function | null} onClick - Click handler.
 * @param {string} [variant='primary'] - Visual variant of the button.
 * @returns {HTMLButtonElement}
 */
export function Button(text, onClick, variant = 'primary') {
    return el('button', {
        class: variant === 'ghost' ? 'btn btn-ghost' : 'btn',
        text,
        onclick: onClick || undefined
    });
}
